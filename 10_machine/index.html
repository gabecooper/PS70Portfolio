DrawGPT: A Drawing Robot

For machines week, our group chose to create DrawGPT, a drawing robot that enables LLMs to draw seamlessly in conversation with a human artist. In theory, this robot would be able to finish drawings, respond to written commands, and even play games… like tic-tac-toe! In actuality, DrawGPT doesn't listen to instructions very well, struggles to draw a circle, but still has a beautiful roof over its head. Here's how it went:

Magically, the machine was built by Friday morning!

So it turns out that "magically" means Aurora spent a week building it, but it felt magical. Come Friday morning, the machine was built exactly according to the kit with the breadboard soldered, the wires braided (Hannah got bored, I think), and the conveyor belt hooked up to the motors. We also spent some time on the finer details like lubricating the bars so there was less friction (which is SUPER important to mention - see img 4). There were countless times when Bobby got a shoutout in the team groupchat during this process, but after the machine was finally built we were all grateful that it was completed for good and that we would never have to touch it again (foreshadowing).

Base Construction:

Naturally, like with any modern SAAS AI wrapper, our key focus was on the packaging. Hence, Hannah prioritized the "UX" of our machine which involved making it sturdy enough such that the calibration between the drawing machine and the Arucos markers would hold even throughout back-and-forth conversations. Furthermore, we also included a clamp that is controlled by our ESP32C3 such that it holds that paper while a session is active and releases afterwards. This was likewise important to maintain the calibration with the Arucos markers. If we were to continue this project by skipping all of our finals, we've decided that we would also like to make the paper attached to a roller such that it is "scrollable" like real ChatGPT.

  // ── Servo update ──────────────────────────────────────────
  void updateServo() {
    if (millis() - lastServoMillis < 15) return;
    lastServoMillis = millis();
  
    if (servoPos < servoTarget)       { servoPos++;  myServo.write(servoPos); }
    else if (servoPos > servoTarget)  { servoPos--;  myServo.write(servoPos); }
  
    if (clampPos < clampTarget)       { clampPos++;  clampServo.write(clampPos); }
    else if (clampPos > clampTarget)  { clampPos--;  clampServo.write(clampPos); }
  }

  // ── Parse WebSocket message ───────────────────────────────
  void parseMessage(const String& msg, uint32_t clientId) {
    Serial.printf("[PARSE] Raw message: \"%s\" from client #%u\n", msg.c_str(), clientId);
  
    int ampIdx  = msg.indexOf('&');
    int dashIdx = msg.indexOf('-');
  
    if (ampIdx == -1) {
      Serial.println("[PARSE] ERROR — missing '&' separator, ignoring message");
      return;
    }
    if (dashIdx == -1) {
      Serial.println("[PARSE] ERROR — missing '-' separator, ignoring message");
      return;
    }
    if (dashIdx < ampIdx) {
      Serial.println("[PARSE] ERROR — '-' appears before '&', bad format");
      return;
    }
  
    String xStr   = msg.substring(0, ampIdx);
    String yStr   = msg.substring(ampIdx + 1, dashIdx);  // parsed but unused
    String penStr = msg.substring(dashIdx + 1);
  
    Serial.printf("[PARSE] Extracted → x=\"%s\"  y=\"%s\" (unused)  pen=\"%s\"\n",
                  xStr.c_str(), yStr.c_str(), penStr.c_str());
  
    int x   = xStr.toInt();
    int pen = penStr.toInt();
  
    if (pen != 0 && pen != 1) {
      Serial.printf("[PARSE] WARNING — pen value %d is not 0 or 1, defaulting to up\n", pen);
      pen = 0;
    }
  
    target_X      = x;
    target_Servo  = (pen == 1) ? 70 : 120;  // 1=down(70°), 0=up(120°)
    pendingClient = clientId;
    newRequest    = true;
  
    Serial.printf("[PARSE] Queued → X:%d  Servo target:%d° (%s)\n",
                  target_X, target_Servo, pen ? "DOWN" : "UP");
  }
  
  // ── WebSocket events ──────────────────────────────────────
  void onWsEvent(AsyncWebSocket* server, AsyncWebSocketClient* client,
                 AwsEventType type, void* arg, uint8_t* data, size_t len) {
    switch (type) {
      case WS_EVT_CONNECT:
        Serial.printf("[WS] Client #%u connected from %s\n",
                      client->id(), client->remoteIP().toString().c_str());
        if (homing || postHoming) {
          client->text("homing");
          Serial.println("[WS] Sent 'homing' — not ready yet");
        } else {
          client->text("ready");
          Serial.println("[WS] Sent 'ready'");
        }
        break;
  
      case WS_EVT_DISCONNECT:
        Serial.printf("[WS] Client #%u disconnected\n", client->id());
        if (client->id() == pendingClient) {
          Serial.println("[WS] WARNING — disconnected client had a pending ACK, clearing");
          pendingClient = 0;
        }
        if (!homing && !postHoming) {
          clampTarget = 20;
          Serial.println("[WS] Plot complete — clamp opening");
        }
        break;
  
      case WS_EVT_DATA: {
        AwsFrameInfo* info = (AwsFrameInfo*)arg;
        Serial.printf("[WS] Data from client #%u — %u bytes\n", client->id(), len);
        if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
          data[len] = 0;
          String msg = (char*)data;
          msg.trim();
          if (homing || postHoming) {
            client->text("homing");
            Serial.println("[WS] Message received but not ready — replied 'homing'");
          } else if (newRequest) {
            Serial.println("[WS] WARNING — new message received while previous motion not dispatched yet");
            parseMessage(msg, client->id());
          } else {
            parseMessage(msg, client->id());
          }
        } else {
          Serial.println("[WS] WARNING — received non-text or fragmented frame, ignoring");
        }
        break;
      }
  
      case WS_EVT_ERROR:
        Serial.printf("[WS] ERROR on client #%u\n", client->id());
        break;
  
      case WS_EVT_PONG:
        Serial.printf("[WS] Pong from client #%u\n", client->id());
        break;
  
      default:
        Serial.println("[WS] Unknown event type");
        break;
    }
  }
  
  // ── Homing ────────────────────────────────────────────────
  void runHoming() {
    static bool x1HomingStarted = false;
    static bool x2HomingStarted = false;
  
    // X2 homes first
    if (!x2Homed) {
      if (!x2HomingStarted) {
        Serial.println("[HOME] X2 homing started — moving toward limit switch D7");
        x2HomingStarted = true;
      }
      if (digitalRead(D7) == LOW) {
        stepper_X2.stop();
        stepper_X2.setCurrentPosition(0);
        x2Homed  = true;
        d7WasHit = true;
        Serial.println("[HOME] X2 limit hit — position zeroed");
      } else {
        stepper_X2.setSpeed(-200);
        stepper_X2.runSpeed();
      }
    }
  
    // X1 homes only after X2 is done
    if (x2Homed && !x1Homed) {
      if (!x1HomingStarted) {
        Serial.println("[HOME] X1 homing started — moving toward limit switch D8");
        x1HomingStarted = true;
      }
      if (digitalRead(D8) == LOW) {
        stepper_X1.stop();
        stepper_X1.setCurrentPosition(0);
        x1Homed  = true;
        d8WasHit = true;
        Serial.println("[HOME] X1 limit hit — position zeroed");
      } else {
        stepper_X1.setSpeed(-200);
        stepper_X1.runSpeed();
      }
    }
  
    if (x1Homed && x2Homed) {
      homing      = false;
      postHoming  = true;
      clampTarget = 60;
      stepper_X1.moveTo(5000);
      Serial.println("[HOME] ✓ Homing complete — clamp closing, moving X1 to start position 5000...");
    }
  }

  // ── Loop ──────────────────────────────────────────────────
  void loop() {
    static unsigned long lastIPPrint = 0;
    if (millis() - lastIPPrint > 5000) {
      lastIPPrint = millis();
      Serial.printf("[STATUS] IP: %s | WiFi: %s | Homing: %s | PostHome: %s | PendingClient: %u\n",
                    WiFi.localIP().toString().c_str(),
                    WiFi.status() == WL_CONNECTED ? "OK" : "LOST",
                    homing ? "yes" : "no",
                    postHoming ? "yes" : "no",
                    pendingClient);
    }
  
    // Phase 1 — homing
    if (homing) {
      runHoming();
      return;
    }
  
    // Phase 2 — post-home travel to start position
    if (postHoming) {
      stepper_X1.run();
      updateServo();
      if (stepper_X1.distanceToGo() == 0) {
        postHoming = false;
        Serial.printf("[HOME] ✓ X1 at start position (%ld) — ready for commands\n",
                      stepper_X1.currentPosition());
        ws.textAll("ready");
      }
      return;
    }
  
    // Phase 3 — normal operation
    checkLimitSwitches();
  
    if (newRequest) {
      Serial.printf("[MOTION] Dispatching → X1+X2 moveTo(%d), servo target %d°\n",
                    target_X, target_Servo);
      stepper_X1.moveTo(target_X);
      stepper_X2.moveTo(target_X);
      servoTarget      = target_Servo;
      newRequest       = false;
      motionDispatched = true;
    }
  
    stepper_X1.run();
    stepper_X2.run();
    updateServo();
  
    bool steppersDone = stepper_X1.distanceToGo() == 0 && stepper_X2.distanceToGo() == 0;
    bool servoDone    = servoPos == servoTarget;
    bool motionDone   = motionDispatched && steppersDone && servoDone;
  
    static bool wasDone = true;
    if (!wasDone && motionDone) {
      Serial.printf("[MOTION] ✓ Motion complete — X1:%ld  X2:%ld  Servo:%d°\n",
                    stepper_X1.currentPosition(),
                    stepper_X2.currentPosition(),
                    servoPos);
    }
    wasDone = motionDone;
  
    if (motionDone && pendingClient != 0) {
      ws.textAll("done");
      Serial.printf("[WS] ✓ Sent 'done' to all clients (was pending #%u)\n", pendingClient);
      pendingClient    = 0;
      motionDispatched = false;
    }
  
    ws.cleanupClients();
  }

Recap:

In the end, by writing "Draw a Circle" on the page, using the tool-calling script (yes, this just means that it used the circle tool and specified the location and size lol), DrawGPT was able to draw an oval! In fairness, my preview script believes that it was a circle so it follows that maybe the motor that we switched in has a different step size? Lastly, another useful anecdote was that on the final presentation day Marc mentioned OmniSVG would likely be a good fit given that it was designed to create SVGs. I do believe that even Gemini has the capacity to "use" the machine somewhat correctly, but I think you'd have to turn the token allowance way up so that it could write code. Nevertheless, I did some research and discovered that LLMs are notoriously pretty bad at drawing, chess, and spatial reasoning given these all have a "precise and structured state" and LLMs aren't reliably deterministic when given hard constraints. With drawing, then, it follows that LLMs with a small token budget are "losing track" of what's where and what they've done so far. Therefore, like with chess, a hybrid model is likely a better choice for this exact problem.
